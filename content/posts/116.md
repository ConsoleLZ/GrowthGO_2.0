u16 POINT_COLOR = 0x0000, BACK_COLOR = 0xFFFF;

void LCD_GPIO_Init()
{
    // 初始拉高SCL和SDA
    SCL_PIN = 1;
    SDA_PIN = 1;
}

// 复位
void LCD_Reset(void)
{
    RES_PIN = 0;
    delay_ms(100);
    RES_PIN = 1;
    delay_ms(100);
}

void SPI_SendData(uint8_t d)
{
    uint8_t i;
    for (i = 0; i < 8; i++)
    {
        if (d & 0x80)
        {
            SDA_PIN = 1;
        }
        else
        {
            SDA_PIN = 0; // 数据线置低
        }
        d <<= 1;

        // 产生时钟上升沿
        SCL_PIN = 0; // 时钟线置低
        delay_us(1);
        SCL_PIN = 1; // 时钟线置高
        delay_us(1);
    }
}

// 发送命令
void LCD_WriteCmd(uint8_t cmd)
{
    DC_PIN = 0;
    SPI_SendData(cmd);
}

// 发送数据

{
    DC_PIN = 1;
    SPI_SendData(d);
}

// 发送16位数据
static void LCD_WriteData16(uint16_t d)
{
    DC_PIN = 1;
    SPI_SendData(d >> 8);
    SPI_SendData(d & 0xFF);
}

// 设置显示区域
void LCD_SetWindow(uint16_t x1, uint16_t y1, uint16_t x2, uint16_t y2)

    LCD_WriteCmd(0x2A); // 列地址设置
    LCD_WriteData(x1 >> 8);
    LCD_WriteData(x1 & 0xFF);
    LCD_WriteData(x2 >> 8);
    LCD_WriteData(x2 & 0xFF);

    LCD_WriteCmd(0x2B); // 行地址设置
    LCD_WriteData(y1 >> 8);
    LCD_WriteData(y1 & 0xFF);
    LCD_WriteData(y2 >> 8);
    LCD_WriteData(y2 & 0xFF);

    LCD_WriteCmd(0x2C); // 写入RAM
}

/**
 * @brief  在指定位置显示一个字符
 * @param  x: 字符左上角x坐标(0~239)
 * @param  y: 字符左上角y坐标(0~239)
 * @param  chr: 要显示的字符(ASCII码)
 * @param  color: 字符颜色
 * @retval 无
 */
void LCD_ShowChar(u16 x, u16 y, u8 chr, u16 color)
{
	u8 *pfont = 0;
	u8 size = 16;
	u8 csize = (size / 8 + ((size % 8) ? 1 : 0)) * (size); // 计算字符点阵所占字节数
    u8 row, col, temp;

	// 获取字符在字模中的偏移量(ASCII码从32开始)
	chr = chr - ' ';
	pfont = (u8 *)&asc2_1608[chr][0];

	// 设置显示窗口为字符区域
	LCD_SetWindow(x, y, x + size - 1, y + size - 1);

	// 逐行逐列绘制字符
	for (row = 0; row < 16; row++) // 16行点阵
	{
		temp = *pfont++;				 // 获取当前行点阵数据
		for (col = 0; col < 8; col++) // 每行8位
		{
			if (temp & (0x80 >> col))
			{ // 从最高位开始判断
				POINT_COLOR = color;
				LCD_DrawPoint(x + size - 1 - col, y + row);
			}
			else
			{
				POINT_COLOR = BACK_COLOR;
				LCD_DrawPoint(x + size - 1 - col, y + row);
			}
		}
	}
}

/**
 * @brief  在指定位置显示字符串
 * @param  x: 字符串左上角x坐标
 * @param  y: 字符串左上角y坐标
 * @param  p: 字符串指针
 * @param  color: 字符颜色
 * @retval 无
 */
void LCD_ShowString(u16 x, u16 y, u8 *p, u16 color)
{
    // 只要字符串没有结束（'\0'），就继续循环
    while (*p)
    {
        // 调用已有的单字符显示函数
        // 注意：这里使用的是你代码中的 16x8 字体，所以宽度是 8
        LCD_ShowChar(x, y, *p, color);
        
        // 更新 x 坐标，指向下一个字符的位置
        // 因为使用的是 8x16 字体，所以 x 每次增加 8
        x += 8; 
        
        // 移动到下一个字符
        p++;
    }
}

/**
 * @brief  在指定位置显示一个汉字
 * @param  x: 汉字左上角x坐标(0~239)
 * @param  y: 汉字左上角y坐标(0~239)
 * @param  index: 汉字在字库中的索引
 * @param  color: 汉字颜色
 * @retval 无
 */
void LCD_ShowChineseChar(u16 x, u16 y, u8 index, u16 color)
{
	u8 temp1, temp2, row, col;
	u8 *pfont = (u8 *)&hanzi1[index * 2][0]; // 获取汉字点阵数据(每个汉字占2行16字节数据)

	// 设置显示窗口为16x16区域
	LCD_SetWindow(x, y, x + 15, y + 15);

	// 逐行绘制汉字(16行)
	for (row = 0; row < 16; row++)
	{
		temp1 = *pfont++; // 获取当前行高字节
		temp2 = *pfont++; // 获取当前行低字节

		// 处理高字节(8位) - 左半部分
		for (col = 0; col < 8; col++)
		{
			if (temp1 & (0x80 >> col))
			{ // 从最高位开始判断
				POINT_COLOR = color;
				LCD_DrawPoint(x + col, y + row); // 修改为从左到右绘制
			}
			else
			{
				POINT_COLOR = BACK_COLOR;
				LCD_DrawPoint(x + col, y + row);
			}
		}

		// 处理低字节(8位) - 右半部分
		for (col = 0; col < 8; col++)
		{
			if (temp2 & (0x80 >> col))
			{ // 从最高位开始判断
				POINT_COLOR = color;
				LCD_DrawPoint(x + 8 + col, y + row); // 修改为从左到右绘制
			}
			else
			{
				POINT_COLOR = BACK_COLOR;
				LCD_DrawPoint(x + 8 + col, y + row);
			}
		}
	}
}

/**
 * @brief  在指定位置显示一幅图像，在使用Image2Lcd取模的时候记得勾选高位在前，不然显示的颜色会是反的
 * @param  x: 图像左上角x坐标
 * @param  y: 图像左上角y坐标
 * @param  width: 图像宽度(像素)
 * @param  height: 图像高度(像素)
 * @param  image: 图像数据指针(每个像素由两个连续的8位数据组成，对应16位RGB565格式)
 * @retval 无
 */
void LCD_ShowImage(u16 x, u16 y, u16 width, u16 height, const u8 *image)
{
    u32 i;
    u32 pixel_count = (u32)width * height; // 计算总像素点数

    // 1. 设置显示窗口
    LCD_SetWindow(x, y, x + width - 1, y + height - 1);

    // 2. 准备写入数据 (DC引脚拉高)
    DC_PIN = 1; 

    // 3. 循环写入像素数据
    // 图片数据通常是RGB565格式，每个像素占2个字节
    // image[i]是高8位，image[i+1]是低8位
    for (i = 0; i < pixel_count * 2; i += 2)
    {
        // 发送高8位
        SPI_SendData(image[i]);
        // 发送低8位
        SPI_SendData(image[i + 1]);
    }
}

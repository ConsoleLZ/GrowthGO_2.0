## 前言
由于Img2LCD生成的取模的数据的数组每一项都是8位，所以在选择16位真彩色后，可能需要一个转换成每一项都是16位的数组的功能，所以我直接写了一个脚本


## 脚本内容
```js
const readline = require('readline');

// 创建命令行读取接口
const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
});

console.log('请输入 Image2Lcd 生成的 C 数组数据 (8位)：');
console.log('（支持单行或多行粘贴，输入完成后停顿 0.5 秒即可自动转换）\n');

let inputText = '';
let timer = null; // 定义一个定时器变量

// 监听每一行的输入
rl.on('line', (line) => {
    // 每次接收到新的一行，就把内容拼接到总字符串中
    inputText += line + '\n';

    // 核心逻辑：清除上一次的定时器，并重新设置一个新的
    if (timer) clearTimeout(timer);
    
    timer = setTimeout(() => {
        // 如果 500 毫秒内没有新的 'line' 事件触发，就认为输入结束了
        processData();
        rl.close(); // 关闭接口，退出程序
    }, 500); // 这里设置等待时间为 500 毫秒
});

// 处理数据的核心函数
function processData() {
    // 1. 清理字符串：移除 0X, 标点符号和空白字符
    let cleaned = inputText
        .replace(/0x/gi, '') // 移除 0x 或 0X
        .replace(/[^0-9a-fA-F]/g, ' ') // 保留十六进制字符，其余替换为空格
        .trim();

    // 2. 分割成数组
    let hexArray = cleaned.split(/\s+/).filter(Boolean); // 过滤掉空字符串

    console.log(`\n--- 转换结果 ---`);
    console.log(`检测到 ${hexArray.length} 个 8位 数据项。`);

    // 检查数据长度是否为偶数
    if (hexArray.length % 2 !== 0) {
        console.log('⚠️ 警告：数据长度为奇数，可能缺少最后一个字节，已自动忽略最后一个字节。');
        hexArray.pop(); // 去掉最后一个不完整的字节
    }

    // 3. 转换为 16位 数组
    let outputArray = [];
    for (let i = 0; i < hexArray.length; i += 2) {
        // Image2Lcd 默认是 "高位在前" (MSB First)
        let highByte = hexArray[i];     // 高8位
        let lowByte = hexArray[i + 1];  // 低8位
        
        // 组合成 16位
        let combined = `0x${highByte.toUpperCase()}${lowByte.toUpperCase()}`;
        outputArray.push(combined);
    }

    console.log(`转换为 ${outputArray.length} 个 16位 数据项。\n`);

    // 4. 格式化输出 (每行 8 个数据)
    let outputString = 'const uint16_t gImage_Converted[] = {\n';
    for (let i = 0; i < outputArray.length; i += 8) {
        let row = outputArray.slice(i, i + 8);
        outputString += '    ' + row.join(', ') + ',\n';
    }
    // 只有当有数据时才去掉最后的逗号和换行
    if (outputArray.length > 0) {
        outputString = outputString.slice(0, -2); 
    }
    outputString += '\n};';

    console.log(outputString);
}

// 防止用户没有任何输入直接 Ctrl+D/Ctrl+Z 退出导致没反应
rl.on('close', () => {
    if (!inputText) {
        console.log('\n未检测到任何输入，程序已退出。');
    }
});
```
